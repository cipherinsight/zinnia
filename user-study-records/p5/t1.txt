use core::u64;
use array::ArrayTrait;

fn main(n: u64) -> u8 {
    if (n > 10000 || n < 1) {
        return 0;
    }
    if (n == 1) {
        return 1;
    }
    let mut i = 2;
    let mut flag = 0;
    loop {
        if (i == n) {
            flag = 1;
            break;
        }
        if (n % i == 0) {
            flag = 0;
            break;
        }
        i = i + 1;
    };
    return flag;
}
