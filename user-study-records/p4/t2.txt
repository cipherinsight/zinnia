fn main(n: u64, ans: u64) {
    if(n==1) {
        assert!(ans == 1, "ans wrong");
        return;
    }
    if(n==2) {
        assert!(ans == 2, "ans wrong");
        return;
    }
    let mut prev: u64 = 1;
    let mut curr: u64 = 2;
    let mut i: u64 = 2;
    while i < n {
        let temp = curr;
        curr = prev + curr;
        prev = temp;
        i = i + 1;
    }
    assert!(curr == ans, "ans wrong");
}
