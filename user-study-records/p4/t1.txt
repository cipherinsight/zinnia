fn main(n: u64, ans: u64) {
    let mut i: u64 = 2;
    if(n == 1) {
        assert!(ans == 0, "ans wrong");
        return;
    }
    while i*i <=n {
        if n % i == 0 {
            assert!(ans == 0, "ans wrong");
            return;
        }
        i = i + 1;
    }
    assert!(ans == 1, "ans wrong");
}
